import pandas as pd
from sklearn.linear_model import LogisticRegression

y = data['default_flag']

# Train logistic regression model
model = LogisticRegression()
model.fit(X, y)

# Predict default probabilities
probs = model.predict_proba(X)[:,1]
print(probs[:5])
